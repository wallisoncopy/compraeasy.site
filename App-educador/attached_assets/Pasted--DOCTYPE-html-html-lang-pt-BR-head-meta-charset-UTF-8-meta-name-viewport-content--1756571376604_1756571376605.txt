<script>
  // Timer simples (fake para simular urgência)
  let minutes = 14;
  let seconds = 42;
  const countdownEl = document.getElementById("countdown");

  function updateCountdown() {
    seconds--;
    if (seconds < 0) {
      seconds = 59;
      minutes--;
    }
    if (minutes < 0) {
      minutes = 14;
      seconds = 42;
    }
    countdownEl.textContent = minutes + " min : " + (seconds < 10 ? "0" : "") + seconds + " seg";
  }

  setInterval(updateCountdown, 1000);
</script>
